import { PatientProfile } from "./types";

export type MurphyPathwayId =
  | "A_high_risk_diagnostic_priority"
  | "B_device_focused"
  | "C_anatomy_dominant"
  | "D_physiology_dominant"
  | "E_comisa_insomnia_first"
  | "F_mild_snoring_lifestyle"
  | "G_complex_mixed";

function anyZoneAtLeast(profile: PatientProfile, n: number): boolean {
  const z = profile.zones;
  return z.nose >= n || z.palate >= n || z.mandible >= n || z.neck >= n;
}

function totalZoneScore(profile: PatientProfile): number {
  const z = profile.zones;
  return z.nose + z.palate + z.mandible + z.neck;
}

export function determineProblemDominance(
  profile: PatientProfile
): PatientProfile["problemDominance"] {
  const anatomicScore = totalZoneScore(profile);
  const e = profile.endotypes;

  const physScore =
    (e?.arousalThresholdLow ? 1 : 0) +
    (e?.loopGainHigh ? 1 : 0) +
    (e?.muscleCompensationPoor ? 1 : 0);

  const pcritBoost = e?.pcritHigh ? 1 : 0;
  const anatomySignal = anatomicScore + pcritBoost;

  if (anatomySignal >= 4 && physScore <= 1) return "anatomy";
  if (physScore >= 2 && anatomySignal <= 2) return "physiology";
  if (anatomySignal >= 3 && physScore >= 2) return "mixed";
  return "uncertain";
}

export function assignMurphyPathway(profile: PatientProfile): MurphyPathwayId {
  const dominance = determineProblemDominance(profile);
  profile.problemDominance = dominance;

  // A: High risk diagnostic priority
  if (
    profile.osaRisk === "high" &&
    profile.comorbidities?.some((c) =>
      ["htn", "afib", "stroke", "heart_failure"].includes(c)
    )
  ) {
    return "A_high_risk_diagnostic_priority";
  }

  // E: COMISA / insomnia-first
  if (
    (profile.insomniaSeverity === "moderate" ||
      profile.insomniaSeverity === "severe") &&
    (profile.osaRisk === "low" || profile.osaRisk === "intermediate")
  ) {
    return "E_comisa_insomnia_first";
  }

  // C: Anatomy-dominant
  if (dominance === "anatomy" && anyZoneAtLeast(profile, 2)) {
    return "C_anatomy_dominant";
  }

  // D: Physiology-dominant
  if (dominance === "physiology") {
    return "D_physiology_dominant";
  }

  // G: Mixed / complex
  if (dominance === "mixed") {
    return "G_complex_mixed";
  }

  // F: Mild snoring lifestyle
  if (profile.osaRisk === "low" && profile.mainComplaint === "snoring") {
    return "F_mild_snoring_lifestyle";
  }

  // B: Default
  return "B_device_focused";
}
